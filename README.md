function daysToYMD(days) {
  const years = Math.floor(days / 365);
  const months = Math.floor((days % 365) / 30);
  const d = days % 30;
  
  const parts = [];
  if (years > 0) parts.push(`${years}Y`);
  if (months > 0) parts.push(`${months}M`);
  if (d > 0 || parts.length === 0) parts.push(`${d}D`);
  
  return parts.join(' ');
}
